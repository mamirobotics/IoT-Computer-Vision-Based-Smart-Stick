#define PCLK_GPIO_NUM     22

httpd_handle_t server = NULL;

//STREAM HANDLER 
static esp_err_t stream_handler(httpd_req_t *req)
{
  camera_fb_t * fb = NULL;

  httpd_resp_set_type(req, "multipart/x-mixed-replace;boundary=frame");

  while (true) {

    fb = esp_camera_fb_get();
    if (!fb) continue;

    uint8_t *jpg_buf = NULL;
    size_t jpg_len = 0;

    //  MEDIUM QUALITY (Best balance)
    bool ok = frame2jpg(fb, 30, &jpg_buf, &jpg_len);

    if (!ok) {
      esp_camera_fb_return(fb);
      continue;
    }

    httpd_resp_send_chunk(req,
      "--frame\r\nContent-Type: image/jpeg\r\n\r\n", 37);

    httpd_resp_send_chunk(req, (const char*)jpg_buf, jpg_len);
    httpd_resp_send_chunk(req, "\r\n", 2);

    free(jpg_buf);
    esp_camera_fb_return(fb);
  }
}

// SERVER 
void startCameraServer()
{
  httpd_config_t config = HTTPD_DEFAULT_CONFIG();
  config.server_port = 80;   //  universal port

  httpd_uri_t uri = {
    .uri       = "/",        // open directly
    .method    = HTTP_GET,
    .handler   = stream_handler,
    .user_ctx  = NULL
  };

  httpd_start(&server, &config);
  httpd_register_uri_handler(server, &uri);
}

//  SETUP 
void setup() {
  Serial.begin(115200);
  Serial.println("Starting...");

  //  WIFI FIRST 
  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid, password);

  Serial.print("Connecting");
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("\n WiFi Connected");
  Serial.print(" IP Address: ");
  Serial.println(WiFi.localIP());

  //  CAMERA CONFIG
  camera_config_t config;
  config.ledc_channel = LEDC_CHANNEL_0;
  config.ledc_timer = LEDC_TIMER_0;
  config.pin_d0 = Y2_GPIO_NUM;
  config.pin_d1 = Y3_GPIO_NUM;
  config.pin_d2 = Y4_GPIO_NUM;
  config.pin_d3 = Y5_GPIO_NUM;
  config.pin_d4 = Y6_GPIO_NUM;
  config.pin_d5 = Y7_GPIO_NUM;
  config.pin_d6 = Y8_GPIO_NUM;
  config.pin_d7 = Y9_GPIO_NUM;
  config.pin_xclk = XCLK_GPIO_NUM;
  config.pin_pclk = PCLK_GPIO_NUM;
  config.pin_vsync = VSYNC_GPIO_NUM;
  config.pin_href = HREF_GPIO_NUM;
  config.pin_sscb_sda = SIOD_GPIO_NUM;
  config.pin_sscb_scl = SIOC_GPIO_NUM;
  config.pin_pwdn = PWDN_GPIO_NUM;
  config.pin_reset = RESET_GPIO_NUM;

  config.xclk_freq_hz = 20000000;

  //  REQUIRED FOR M21-45
  // OV4056 is origianl camera above is copy of this
 // cannot handle the high pixel than I try to run smoothly with medium results
  config.pixel_format = PIXFORMAT_RGB565;

  //  MEDIUM PERFORMANCE SETTINGS
  config.frame_size = FRAMESIZE_QVGA;  // 320x240
  config.fb_count = 1;

  //  INIT CAMERA 
  if (esp_camera_init(&config) != ESP_OK) {
    Serial.println(" Camera Init Failed");
    return;
  }

  Serial.println(" Camera Ready");

  //  START SERVER 
  startCameraServer();

  Serial.println(" Stream Ready!");
  Serial.print(" Open: http://");
  Serial.println(WiFi.localIP());
}
